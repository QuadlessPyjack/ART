43 void gapiLoadCommand(DrawCmd dc)
44 {
45  uint8_t sId = dc.y >> 4; // Sector height is currently 16
46  if (sId != 0)
47  {
48  dc.y = dc.y % 16;
49  if (dc.y < 4) return; // graphic overlaps two sectors - not supported
50  }
51 
52  if (DRAW_COMMAND_BUFFER[sId].next == NULL)
53  {
54  // No command assigned to sector. Allocate and assign.
55 
56  DrawCmd *cmd = malloc(sizeof(dc));
57  *cmd = dc;
58  DRAW_COMMAND_BUFFER[sId].next = cmd;
59 
60  cmd->next = NULL;
61  cmd->prev = &DRAW_COMMAND_BUFFER[sId];
62 
63  return;
64  }
65 
66  DrawCmd *first = malloc(sizeof(DrawCmd));
67  *first = *((DrawCmd*)DRAW_COMMAND_BUFFER[sId].next);
68 
69  free(DRAW_COMMAND_BUFFER[sId].next);
70 
71  DrawCmd *new = malloc(sizeof(dc));
72  *new = dc;
73 
74  DRAW_COMMAND_BUFFER[sId].next = new; // void* can point to anything, cast not needed
75  new->next = first;
76  new->prev = &DRAW_COMMAND_BUFFER[sId];
77 
78  first->prev = new;
79 }

108 void gapiUpdate()
109 {
110  for (uint8_t idx = 0; idx < DRAW_BUFFER_SIZE; ++idx)
111  {
112  DrawCmd *dcPtr = DRAW_COMMAND_BUFFER[idx].next;
113  DrawCmd *prev = NULL;
114  DrawCmd *toBeRemoved = NULL;
115 
116  if (dcPtr == NULL)
117  {
118  continue;
119  }
120 
121  if (CURRENT_DRAW_SECTOR != idx)
122  {
123  CURRENT_DRAW_SECTOR = idx;
124  setDirtySector(idx);
125  setYSector();
126  }
127 
128  do
129  {
130  interpretDrawCommand(*dcPtr);
131  // Get the pointers we need to free this node
132  toBeRemoved = dcPtr;
133  prev = dcPtr->prev;
134 
135  // Advance pointer to next node
136  dcPtr = dcPtr->next;
137 
138  if (dcPtr == NULL)
139  {
140  prev->next = NULL;
141  free(toBeRemoved);
142  break;
143  }
144 
145  // Update links
146  prev->next = dcPtr;
147  dcPtr->prev = prev;
148 
149  // Free node
150  free(toBeRemoved);
151  } while(dcPtr->next != NULL);
152  blit();
153  }
154 }
